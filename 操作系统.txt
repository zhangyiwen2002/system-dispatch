//发送 
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<unistd.h>
#include<sys/types.h>
#include<sys/shm.h>
main()
{
	int shmid;
	char *viraddr;
	char buffer[BUFSIZ];
	
	shmid=shmget(1234,BUFSIZ,0666|IPC_CREAT);
	viraddr=(char*)shmat(shmid,0,0);
	
	while(1)
	{
		puts("Enter some text:");
		fgets(buffer,BUFSIZ,stdin);
		strcat(viraddr,buffer);
		if(strncmp(buffer,"end",3)==0)
		break;
	}
	shmdt(viraddr);
	exit(0);
 } 


//接收 
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<unistd.h>
#include<sys/types.h>
#include<sys/shm.h>
main()
{
	int shmid;
	char *viraddr;

	shmid=shmget(1234,BUFSIZ,0666|IPC_CREAT);
	viraddr=(char*)shmat(shmid,0,0);
	
	printf("Your message is:\n%s",viraddr);
	shmdt(viraddr);
	shmctl(shmid,IPC_RMID,0);
	exit(0);
 } 


//软中断 
#include<stdio.h>
#include<unistd.h>
#include<stdlib.h>
#include<signal.h>

void int_func(int sig);
int k;

void int_func(int sig)
{
	k=0;
}

main()
{
	int p,shmid,running=1;
	
	char *viraddr1，*viraddr2;
	char buffer[BUFSIZ];
	signal(SIGUSR1,int_func);
	k=1;
	shmid=shmget(1234,BUFSIZ,0666|IPC_CREAT);
	
	while((p=fork())==-1);
	if(p==0)
	{
		while(k==1);
		viraddr2=shmat(shmid,0,0)
		printf("Your message is:\n%s",viraddr2);
		shmdt(viraddr2);
		exit(0);
	}
	else
	{
		viraddr1=shmat(shmid,0,0);
		while(running)
		{
			puts("Enter some text:");
			fgets(buffer,BUFSIZ,stdin);
			strcat(viraddr,buffer);
			if(strncmp(buffer,"end",3)==0);
			running=0;
		}
		shmdt(viraddr1);
		kill(p,SIGUSR1);
		shmctl(shmid,IPC_RMID,0);
		exit(0);
	}
}
